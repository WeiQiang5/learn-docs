5.函数 ​

标准的函数写法

参数 参数类型 返回的类型

go
func getSum(n1 int,n2 int) int{
    sum := n1 + n2
    return sum
}

func main(){
    res := getSum(1,2)
    fmt.Println(res)
}

实参：给函数实际传递的参数，例如上面的1，2 形参：形式上的参数，例如上面的n1和n2

  1. 返回值 golang支持多个返回值
go
func getRes(n1,n2 int) (int,int){
    sum := n1 + n2
    difference := n1-n2
    return sum,difference
}
  1. 返回值命名
go
func getRes(n1,n2 int) (sum,difference int){
    sum = n1 + n2
    difference = n1-n2
    return
}
  1. 函数也是一种数据类型 函数不是变量，但是他也存在与内存中，存在内存地址 函数名的本质就是一个指向其函数内存地址的指针变量
go
func getRes(n1,n2 int)(sum,difference int){
    sum = n1 + n2
    difference = n1 - n2
    return
}
func main(){
    fmt.Print("getRes is %v,type is %T",getRes,getRes)
}

所以，我们也可以以声明变量的形式声明函数

go
var getRes = func(n1,n2 int)(sum,difference int){
    sum = n1 + n2
    difference = n1 - n2
    return
}
a,b := getRes(1,2)
fmt.Println(a,b)

getRes对应的那个函数也叫匿名函数 4. 立即调用的匿名函数

go
a,b := func(n1,n2 int)(sum,difference int){
    sum = n1 + n2
    difference = n1 - n2
    return
}(1,2)
fmt.Println(a,b)

6.指针 ​

go
func add (n int){
    n++
    fmt.Println("add函数结束时: ",n)
}

func main(){
    var y = 2024
    add(y)
    fmt.Println("调用add函数之后")
}

上面可以看到函数调用之后，y的值还是2022 这就是说明函数传递默认是值传递 我们可以使用&可以获取某个变量的内存地址 用*获取到内存地址所对应的值

func add (n *int){
    *n++
    fmt.Println("add函数结束"，n,*n)
}
func main(){
    var y = 2022
    var yy = &y
    add(yy)
    fmt.Println("调用add函数之后")
}

7.defer ​

1.延迟执行的函数会被压入栈中，return后按照先进后出的顺序调用 2.延迟执行的函数其参数会立即求值

go
package function

import "fmt"

func deferUtil() func(int) int {
	i := 0
	return func(n int) int {
		fmt.Printf("本次调用接收到n=%v\n", n)
		i++
		fmt.Printf("匿名工具函数第%v次被调用\n", i)
		return i
	}
}
func testDefer() int {
	f := deferUtil()
	defer f(1)
	return f(2)
}

func MyDefer() {
	testDefer()
}

可用于函数异常捕获

继承

go
package myStruct

import "fmt"

type User struct {
	Name     string
	Age      int
	password string
}

type Account struct {
	User
	money float32
	Name  string
}

func MyStruct() {
	var ac Account = Account{
		money: 21.9,
		User: User{
			Name:     "王武",
			Age:      21,
			password: "978345",
		},
		Name: "账户1",
	}
	fmt.Printf("ac %#v\n,%T\n", ac, ac)

	var u1 = User{
		Name:     "lisa",
		Age:      22,
		password: "123",
	}
	var ac1 Account = Account{
		money: 24.5,
		User:  u1,
	}
	fmt.Printf("ac1 %#v, %T\n", ac1, ac1)

	fmt.Println(ac.money)
	fmt.Println(ac.Name)
	fmt.Println(ac.User.Name)
	fmt.Println(ac.password)
}

方法

go
package myStruct

import "fmt"

type User struct {
	Name     string
	Age      int
	password string
}

func (u User) PrintName() string {
	u.Name = "张三"
	fmt.Println("printName方法:==>", u)
	fmt.Printf("printName方法内部:%p \n", &u)
	return u.Name
}

func MyStruct() {
	user := User{"王二麻子", 23, "sasdsa"}
	fmt.Printf("main : %p\n", &user)
	name := user.PrintName()
	fmt.Println(name, user)
}
